# Diagnose a PartDesign AdditiveHelix / helical sweep path feature.
# Reports axis, placement, profile attachment, handedness, pitch, height,
# and resulting solid metrics. Read-only JSON.
import json as _json

_name = $helix_name
_obj = _doc.getObject(_name)
if _obj is None:
    raise RuntimeError('Helix feature not found: ' + _name)

_props = set(getattr(_obj, 'PropertiesList', []))


def _safe(attr, default=None):
    try:
        return getattr(_obj, attr)
    except Exception:
        return default


def _vec(v):
    if v is None:
        return None
    try:
        return {'x': float(v.x), 'y': float(v.y), 'z': float(v.z)}
    except Exception:
        return None


def _link_ref(val):
    if val is None:
        return None
    try:
        if isinstance(val, (list, tuple)) and val:
            first = val[0]
            if isinstance(first, (list, tuple)) and first:
                obj, subs = first[0], (first[1] if len(first) > 1 else [])
                return {
                    'object': getattr(obj, 'Name', str(obj)),
                    'subs': list(subs) if subs else [],
                }
            return {'object': getattr(first, 'Name', str(first)), 'subs': []}
        return {'object': getattr(val, 'Name', str(val)), 'subs': []}
    except Exception:
        return {'raw': str(val)}


def _f(attr):
    try:
        return float(_safe(attr))
    except Exception:
        return _safe(attr)


_pl = _safe('Placement')
_placement = None
if _pl is not None:
    _placement = {
        'base': _vec(_pl.Base),
        'rotation': str(_pl.Rotation),
    }

_axis = None
for _cand in ('Axis', 'ReferenceAxis', 'Spine'):
    if _cand in _props:
        _axis = _link_ref(_safe(_cand))
        if _axis:
            break
if _axis is None and 'Axis' in _props:
    _axis = _vec(_safe('Axis'))

_profile = None
for _cand in ('Profile', 'Sketch', 'Section'):
    if _cand in _props:
        _profile = _link_ref(_safe(_cand))
        if _profile:
            break

_attach = None
for _cand in ('AttachmentSupport', 'Support', 'MapMode'):
    if _cand in _props:
        if _cand == 'MapMode':
            _attach = {'map_mode': str(_safe(_cand))}
        else:
            _attach = _link_ref(_safe(_cand))
        if _attach:
            break

_left = None
if 'LeftHanded' in _props:
    _left = bool(_safe('LeftHanded'))
elif 'Handedness' in _props:
    _left = str(_safe('Handedness')).lower() in ('left', 'lefthanded', 'left-handed')

_reversed = bool(_safe('Reversed', False)) if 'Reversed' in _props else None

_shape = _safe('Shape')
_null = True
_volume = None
_bbox = None
if _shape is not None:
    try:
        _null = bool(_shape.isNull())
    except Exception:
        _null = True
    if not _null:
        try:
            _volume = float(_shape.Volume)
        except Exception:
            _volume = None
        try:
            _bb = _shape.BoundBox
            _bbox = {
                'xmin': float(_bb.XMin), 'ymin': float(_bb.YMin), 'zmin': float(_bb.ZMin),
                'xmax': float(_bb.XMax), 'ymax': float(_bb.YMax), 'zmax': float(_bb.ZMax),
            }
        except Exception:
            _bbox = None

print(_json.dumps({
    'ok': True,
    'helix': _name,
    'type_id': getattr(_obj, 'TypeId', ''),
    'axis': _axis,
    'placement': _placement,
    'profile': _profile,
    'attachment': _attach,
    'left_handed': _left,
    'reversed': _reversed,
    'pitch': _f('Pitch') if 'Pitch' in _props else None,
    'height': _f('Height') if 'Height' in _props else None,
    'radius': _f('Radius') if 'Radius' in _props else None,
    'turns': _f('Turns') if 'Turns' in _props else None,
    'angle': _f('Angle') if 'Angle' in _props else None,
    'shape_null': _null,
    'volume': _volume,
    'bbox': _bbox,
}))
