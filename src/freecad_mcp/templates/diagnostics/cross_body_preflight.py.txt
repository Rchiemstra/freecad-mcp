# I6 cross-body pre-flight: warn at creation time when a datum/binder attaches
# to a support in a DIFFERENT body that has a non-identity placement — the P1
# cross-body attachment-drop risk. Emits a `__PREFLIGHT_WARN__` sentinel that
# _run_json_code strips out and surfaces as a warning, keeping the JSON clean.
import json as _pf_json

_pf_obj = _doc.getObject($obj_name)
_pf_warns = []


def _pf_parent_body(_o):
    if _o is None:
        return None
    _seen = set()

    def _walk(_x):
        if _x is None or id(_x) in _seen:
            return None
        _seen.add(id(_x))
        if getattr(_x, 'TypeId', '') == 'PartDesign::Body':
            return _x
        for _p in getattr(_x, 'InList', []) or []:
            _r = _walk(_p)
            if _r is not None:
                return _r
        return None

    return _walk(_o)


def _pf_placement_identity(_pl):
    try:
        return _pl.isIdentity()
    except Exception:
        try:
            return (float(_pl.Base.x) == 0 and float(_pl.Base.y) == 0
                    and float(_pl.Base.z) == 0
                    and float(_pl.Rotation.Q[0]) == 0
                    and float(_pl.Rotation.Q[1]) == 0
                    and float(_pl.Rotation.Q[2]) == 0)
        except Exception:
            return True


def _pf_supports(_o):
    def _refs(_sup):
        if _sup is None:
            return []
        try:
            return [s for s in list(_sup.getValues()) if s is not None]
        except Exception:
            pass
        try:
            _items = list(_sup)
        except Exception:
            return []
        _objs = []
        for _item in _items:
            try:
                _objs.append(_item[0] if isinstance(_item, tuple) else _item)
            except Exception:
                pass
        return [s for s in _objs if s is not None]

    _found = _refs(getattr(_o, 'AttachmentSupport', None))
    if not _found:
        _found = _refs(getattr(_o, 'Support', None))
    return _found


if _pf_obj is not None:
    _pf_obj_body = _pf_parent_body(_pf_obj)
    for _pf_sup in _pf_supports(_pf_obj):
        _pf_sup_body = _pf_parent_body(_pf_sup)
        if (_pf_sup_body is not None and _pf_obj_body is not None
                and _pf_sup_body.Name != _pf_obj_body.Name
                and not _pf_placement_identity(_pf_sup_body.Placement)):
            _pf_warns.append({
                'datum': _pf_obj.Name,
                'datum_body': _pf_obj_body.Name,
                'support': _pf_sup.Name,
                'support_body': _pf_sup_body.Name,
                'message': (
                    'Cross-body attachment: ' + _pf_obj.Name + ' in body '
                    + _pf_obj_body.Name + ' attaches to ' + _pf_sup.Name
                    + ' in body ' + _pf_sup_body.Name + ', which has a '
                    + 'non-identity placement. FreeCAD may drop the source '
                    + 'body placement (P1). Consider preview_attachment, or '
                    + 'use an identity-placement intermediate body.'
                ),
            })

print('__PREFLIGHT_WARN__' + _pf_json.dumps(_pf_warns))
