    raise RuntimeError('Assembly not found: ' + _assembly_name)

    _is_assembly = _assembly.isDerivedFrom('Assembly::AssemblyObject')

    _is_assembly = False

    raise RuntimeError('Object is not an Assembly::AssemblyObject: ' + _assembly_name)

_component = _doc.getObject(_component_name)
if not _component:
    raise RuntimeError('Component not found: ' + _component_name)

_joint = Assembly.createGroundedJoint(
    _assembly,
    _component,
    label=_label,
    recompute=_recompute,
)
