import FreeCAD, json
_doc = FreeCAD.getDocument($doc_name)
if not _doc: raise RuntimeError($doc_missing)
_obj = _doc.getObject($object_name)
if not _obj: raise RuntimeError('Object not found: ' + $object_name)
_prop = $prop_path
try:
    if hasattr(_obj, 'clearExpression'):
        _obj.clearExpression(_prop)
    else:
        _obj.setExpression(_prop, None)
except Exception as _e:
    print(json.dumps({
        'ok': False,
        'error': 'expression_error',
        'object': _obj.Name,
        'prop_path': _prop,
        'message': str(_e),
        'type': type(_e).__name__,
    }))
    raise RuntimeError('Failed to clear expression on ' + _obj.Name + '.' + _prop + ': ' + str(_e))
_doc.recompute()
print(json.dumps({'ok': True, 'object': _obj.Name, 'prop_path': _prop, 'cleared': True}))
