import FreeCAD, Part, math
_doc = FreeCAD.getDocument($doc_name)
_sk = _doc.getObject($sketch_name)
if not _sk: raise RuntimeError('Sketch not found')
_circ = Part.Circle(FreeCAD.Vector($cx,$cy,0),FreeCAD.Vector(0,0,1),$radius)
_idx = _sk.addGeometry(Part.ArcOfCircle(_circ,math.radians($start_angle),math.radians($end_angle)),$construction)

print('geometry_index=' + str(_idx))
